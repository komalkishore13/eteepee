import mongoose, { Document, Model, Schema, CallbackError } from "mongoose";
import bcrypt from "bcryptjs";

// ─── Types ────────────────────────────────────────────────────────────────

export interface IUser extends Document {
  username: string;
  email: string;
  password: string;        // hash — never returned by default
  avatar: string;
  createdAt: Date;
  updatedAt: Date;

  /** Returns true when the plain-text candidate matches the hash. */
  comparePassword(candidate: string): Promise<boolean>;
}

// ─── Schema ───────────────────────────────────────────────────────────────

const UserSchema = new Schema<IUser>(
  {
    username: {
      type: String,
      required: [true, "Username is required"],
      trim: true,
      minlength: [3, "Username must be at least 3 characters"],
      maxlength: [30, "Username cannot exceed 30 characters"],
      match: [
        /^[a-zA-Z0-9_]+$/,
        "Username can only contain letters, numbers, and underscores",
      ],
    },
    email: {
      type: String,
      required: [true, "Email is required"],
      unique: true,
      trim: true,
      lowercase: true,
      match: [/^\S+@\S+\.\S+$/, "Please provide a valid email address"],
    },
    password: {
      type: String,
      required: [true, "Password is required"],
      minlength: [8, "Password must be at least 8 characters"],
      // ✅ select: false — excluded from ALL queries by default
      select: false,
    },
    avatar: {
      type: String,
      default: "",
    },
  },
  {
    timestamps: true,
    // ✅ toJSON transform — strips password even if select("+password") was used
    toJSON: {
      transform(_doc, ret) {
        delete ret.password;
        delete ret.__v;
        return ret;
      },
    },
  }
);

// ─── Indexes ──────────────────────────────────────────────────────────────
UserSchema.index({ username: 1 });

// ─── Middleware ───────────────────────────────────────────────────────────

// ✅ Pre-save hook — impossible to save a plain-text password
UserSchema.pre<IUser>("save", async function (next) {
  if (!this.isModified("password")) return next(); // ← skip unrelated saves
  try {
    this.password = await bcrypt.hash(this.password, 12); // cost factor 12
    next();
  } catch (err) {
    next(err as CallbackError);
  }
});

// ─── Instance methods ─────────────────────────────────────────────────────

// ✅ comparePassword — no bcrypt import needed in route files
UserSchema.methods.comparePassword = function (
  candidate: string
): Promise<boolean> {
  return bcrypt.compare(candidate, this.password);
};

// ─── Model ────────────────────────────────────────────────────────────────

const User: Model<IUser> =
  mongoose.models.User || mongoose.model<IUser>("User", UserSchema);

export default User;
✅ Why select: false matters
Without it, User.findOne({ email }) silently returns the bcrypt hash alongside every other field. Any res.json(user) leaks it. With select: false, the field is excluded unless you explicitly request .select("+password") — and only the login route should ever do that.
✅ Why the pre-save hook is safer than manual hashing
If hashing lives in the route, a second route (password reset, admin panel) can forget it. The hook makes it structurally impossible to persist a plain-text password. The isModified("password") guard ensures that saving an avatar or username update doesn't re-hash an already-hashed string.
⚡ toJSON is belt-and-suspenders
select: false guards at the query level. toJSON guards at serialisation. If a future dev calls .select("+password") legitimately and then accidentally passes the document to res.json(), the transform strips it before the wire.
The login route dropped its own bcrypt import and the dual-branch null-check. It now delegates comparison to user.comparePassword() and uses a single shared error response object to prevent user-enumeration.

import { NextRequest, NextResponse } from "next/server";
import bcrypt from "bcryptjs";     // ← manual import
import { connectDB } from "@/lib/db";
import { signToken } from "@/lib/jwt";
import User from "@/models/User";

export async function POST(request: NextRequest) {
  try {
    const body = await request.json();
    const { email, password } = body;

    if (!email || !password) {
      return NextResponse.json(
        { success: false, message: "Email and password are required" },
        { status: 400 }
      );
    }

    await connectDB();

    // No .select("+password") — password excluded by select:false
    // (This would have been a bug in the old schema)
    const user = await User.findOne({ email: email.toLowerCase().trim() });

    // ❌ Separate branches leak whether the email exists
    if (!user) {
      return NextResponse.json(
        { success: false, message: "Invalid email or password" },
        { status: 401 }
      );
    }

    // ❌ bcrypt called directly in the route
    const isPasswordValid = await bcrypt.compare(password, user.password);
    if (!isPasswordValid) {
      return NextResponse.json(
        { success: false, message: "Invalid email or password" },
        { status: 401 }
      );
    }

    const token = signToken({
      userId: user._id.toString(),
      email: user.email,
      username: user.username,
    });

    const response = NextResponse.json(
      {
        success: true,
        message: "Signed in successfully",
        data: {
          userId: user._id.toString(),
          username: user.username,
          email: user.email,
        },
      },
      { status: 200 }
    );

    response.cookies.set("auth_token", token, {
      httpOnly: true,
      secure: process.env.NODE_ENV === "production",
      sameSite: "lax",
      maxAge: 60 * 60 * 24 * 7,
      path: "/",
    });

    return response;
  } catch (error) {
    console.error("[LOGIN_ERROR]", error);
    return NextResponse.json(
      { success: false, message: "Something went wrong. Please try again." },
      { status: 500 }
    );
  }
}

❌ Old: user-enumeration vulnerability
Even though both branches returned the same message string, they were separate Response objects. A subtle timing difference — no DB hit vs. bcrypt (~300ms) — could let an attacker probe which emails are registered.
✅ New: single response variable
const invalid = NextResponse.json(...) is declared once before both checks. Both branches return invalid — same object reference, same timing path.
⚡ Why .select("+password") is required
Adding select: false to the schema means this query returns null for the password field unless you opt in. comparePassword() would then compare against undefined and always return false. The .select("+password") here is intentional and scoped to this one route.
The register route dropped all manual bcrypt logic. It now passes the plain-text password directly to User.create() — the pre-save hook hashes it before the document ever reaches MongoDB.

import bcrypt from "bcryptjs";      // ← manual bcrypt import
import { connectDB } from "@/lib/db";
import { signToken } from "@/lib/jwt";
import User from "@/models/User";

export async function POST(request: NextRequest) {
  try {
    const body = await request.json();
    const { username, email, password } = body;

    if (!username || !email || !password) {
      return NextResponse.json(
        { success: false, message: "All fields are required" },
        { status: 400 }
      );
    }

      return NextResponse.json(
        { success: false, message: "Password must be at least 8 characters" },
        { status: 400 }

      username: username.trim(),
      email: email.toLowerCase().trim(),

    });

    const token = signToken({
      userId: user._id.toString(),
      email: user.email,
      username: user.username,
    });

    const response = NextResponse.json(
      {
        success: true,
        message: "Account created successfully",
        data: {
          userId: user._id.toString(),
          username: user.username,
          email: user.email,
        },
      },
      { status: 201 }
    );

    response.cookies.set("auth_token", token, {
      httpOnly: true,
      secure: process.env.NODE_ENV === "production",
      sameSite: "lax",
      maxAge: 60 * 60 * 24 * 7,
      path: "/",

    if (error instanceof Error && error.message.includes("validation")) {

        { status: 400 }

    }
    return NextResponse.json(
      { success: false, message: "Something went wrong. Please try again." },
      { status: 500 }

❌ Old: duplicated bcrypt calls are a maintenance trap
If you later need to increase the cost factor from 12 to 14, you must find and update every route that calls bcrypt.hash(). Miss one and you have inconsistently hashed passwords in production.
✅ New: single source of truth for hashing policy
The cost factor, the salt rounds, the timing — all live inside the pre-save hook. Every route that calls User.create() or user.save() gets correct hashing for free, including future routes (password reset, OAuth merge, etc.).
⚡ Is it safe to pass plain-text to User.create()?
Yes — User.create() calls new User(data).save() internally. The pre-save hook fires before the document hits the wire. The plain-text string never reaches MongoDB; only the hash does.
